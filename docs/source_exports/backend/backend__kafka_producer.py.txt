import os
import json
import logging
import threading
from concurrent.futures import ThreadPoolExecutor
from datetime import datetime, timezone

# 📝 Kafka Producer Module: The "Real-time Auditor"
# 
# This module handles the background logging of chat conversations to Apache Kafka.
# It uses a "Fire and Forget" pattern with a ThreadPoolExecutor to ensure that
# logging never delays the user's experience in the FastAPI chat route.

logger = logging.getLogger(__name__)

# --- Kafka Configuration ---
KAFKA_BOOTSTRAP_SERVERS = os.getenv("KAFKA_BROKER", "localhost:9092")
KAFKA_TOPIC = "chat_logs"

# --- Shared Global Resources ---
_producer = None
_producer_lock = threading.Lock() 

# Executor handles background tasks off the main thread
_executor = ThreadPoolExecutor(max_workers=4, thread_name_prefix="kafka-audit")

def _on_send_success(record_metadata):
    """Callback triggered when Kafka confirms message receipt."""
    logger.debug(f"✅ Log delivered to {record_metadata.topic} partition {record_metadata.partition}")

def _on_send_error(excp):
    """Callback triggered if the Kafka broker rejects the message."""
    logger.error(f"❌ Kafka Delivery Error: {excp}")

def _get_producer():
    """
    Lazily connects to the Kafka Broker with fault-tolerant settings.
    """
    global _producer
    if _producer is not None:
        return _producer

    with _producer_lock:
        if _producer is not None:
            return _producer
        
        try:
            from kafka import KafkaProducer
            
            # Production-grade settings:
            # - acks='all': Highest durability (all replicas must confirm)
            # - retries=5: Automatic retry on transient failures
            # - request_timeout_ms=5000: Fail fast if broker is slow
            _producer = KafkaProducer(
                bootstrap_servers=KAFKA_BOOTSTRAP_SERVERS,
                value_serializer=lambda v: json.dumps(v).encode("utf-8"),
                acks="all",
                retries=5,
                request_timeout_ms=5000,
                max_block_ms=2000, # Don't block the calling thread for long
                api_version=(2, 5, 0),
            )
            logger.info("📡 Kafka System: Connected to cluster at %s", KAFKA_BOOTSTRAP_SERVERS)
        except Exception as e:
            logger.error("🚨 Kafka System: Unavailable. Audit trail will be missing: %s", e)
            _producer = None
            
    return _producer

def send_log(user_id: str, message: str, response: str, intent: str = "unknown", sentiment: float = 0.0):
    """
    Dispatches a log record to the background execution pool.
    """
    _executor.submit(_send_sync, user_id, message, response, intent, sentiment)

def _send_sync(user_id, message, response, intent, sentiment):
    """
    Internal delivery logic executed in the 'kafka-audit' thread pool.
    """
    producer = _get_producer()
    if not producer:
        return

    payload = {
        "user_id": user_id,
        "message": message,
        "response": response,
        "intent": intent,
        "sentiment": sentiment,
        "timestamp": datetime.now(timezone.utc).isoformat(),
        "origin": "fastapi-gateway-v2"
    }

    try:
        # Use callbacks for production monitoring
        future = producer.send(KAFKA_TOPIC, payload)
        future.add_callback(_on_send_success)
        future.add_errback(_on_send_error)
    except Exception as e:
        logger.error("❌ Kafka Producer: Dispatch failure: %s", e)

def close_producer():
    """Graceful shutdown ensuring all pending messages are flushed."""
    global _producer
    if _producer:
        logger.info("🧹 Kafka System: Flushing pending logs before shutdown...")
        try:
            _producer.flush(timeout=5)
            _producer.close(timeout=5)
        except Exception as e:
            logger.warning(f"⚠️ Error during Kafka shutdown: {e}")
        _producer = None
    _executor.shutdown(wait=True)
